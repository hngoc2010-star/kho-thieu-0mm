<script>

document.querySelectorAll("input").forEach(i=>{

i.addEventListener("input",calc)
i.addEventListener("blur",formatInput)

})

function number(id){

return parseFloat(
document.getElementById(id).value.replace(/,/g,'')
)||0

}

function format(x){

return Number(x).toLocaleString("en-US",{maximumFractionDigits:2})

}

function formatInput(e){

let v=e.target.value.replace(/,/g,'')

if(v!==""){

e.target.value=Number(v).toLocaleString("en-US")

}

}

function calc(){

let thickness=number("thickness")

let width=number("width")
let length=number("length")

let bundle=number("bundle")
let pcsBundle=number("pcsBundle")

let usd=number("usd")

let exwpcs=number("exwpcs")
let exwcbm=number("exwcbm")

let logistics=number("logistics")

let cfr=number("cfr")

let pcs=bundle*pcsBundle

let cbmPc=(thickness/1000)*(width/1000)*(length/1000)

let cbmBuy=pcs*cbmPc

let cbmSell=pcs*cbmPc

let cbmDiff=cbmSell-cbmBuy

let exwTotal=0

if(exwpcs>0){

exwTotal=pcs*exwpcs

}

else if(exwcbm>0){

exwTotal=cbmBuy*exwcbm

}

let totalCfr=exwTotal+logistics

let selling=cbmSell*cfr*usd

let profit=Math.round(selling-totalCfr)

document.getElementById("pcs").innerText=format(pcs)

document.getElementById("buy").innerText=format(cbmBuy)

document.getElementById("sell").innerText=format(cbmSell)

document.getElementById("diff").innerText=format(cbmDiff)

let p=document.getElementById("profit")

p.innerText=format(profit)

p.className=profit>=0?"blue":"red"

}

if("serviceWorker" in navigator){

navigator.serviceWorker.register("service-worker.js")

}
